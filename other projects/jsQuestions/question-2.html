//12 is the multiple that will by multiplied by the variable multiplyBy

var multiple = 12;

var multiplyBy;

// adding quotations around the * symbol before the equal sign in the equation
//allows multiple and multiplyBy to be displayed as two different numbers with a
//multiplication symbol in between

for(multiplyBy = 1; multiplyBy < 13; multiplyBy++){
	document.write( multiple + “ * “ + multiplyBy = multiple * multiplyBy) ;
}
